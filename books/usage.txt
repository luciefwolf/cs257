books - a program that displays information about the books in a small database

SYNOPSIS: 
	python3 books.py searchAuthor "key phrase"
	python3 books.py searchTitle "key phrase" [-t | -y]
	python3 books.py range startYear endYear
	python3 books.py help

OPTIONS: 
	-t, --title: sort by book titles alphabetically
	-y, --year: sort by publication year
